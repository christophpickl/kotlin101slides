\frame{\frametitle{Some Downsides} 
\pause
\begin{itemize}[<+->]
    \item Still pretty \textbf{young}, but things are getting better
    \begin{itemize}
        \item Some language features missing
        \item Unsatisfying tool support (static code analysis)
    \end{itemize}
    \item Final by default is good, but bad for \textbf{mocking}
    \item \texttt{kotlinc} is significantly \textbf{slower} than \texttt{javac}
\end{itemize}
}

\frame{\frametitle{Summary}
\pause
\begin{itemize}[<+->]
	\item Kotlin feels like \textbf{Java 2.0} (\textit{Java on steroids})
	\item It's backed by a \textbf{company} and still \textbf{open source}
	\item Pretty awesome \textbf{community} work
	\item Version \textbf{1.0} just has been released (begin 2016)
	\item \textbf{Roadmap} promises lots of improvements
\end{itemize}
}
